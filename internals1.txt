// Insertion Sort 1
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[n];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    int last = a[n - 1];
    int i = n - 2;

    while (i >= 0 && a[i] > last) {
        a[i + 1] = a[i];
        for (int j = 0; j < n; j++) printf("%d ", a[j]);
        printf("\n");
        i--;
    }

    a[i + 1] = last;
    for (int j = 0; j < n; j++) printf("%d ", a[j]);
    printf("\n");

    return 0;
}

// Insertion Sort 2
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[n];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    for (int i = 1; i < n; i++) {
        int key = a[i], j = i - 1;

        while (j >= 0 && a[j] > key) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;

        for (int k = 0; k < n; k++) printf("%d ", a[k]);
        printf("\n");
    }

    return 0;
}

// Compare the Triplets
#include <stdio.h>

int main() {
    int a[3], b[3], as = 0, bs = 0;

    for (int i = 0; i < 3; i++) scanf("%d", &a[i]);
    for (int i = 0; i < 3; i++) scanf("%d", &b[i]);

    for (int i = 0; i < 3; i++) {
        if (a[i] > b[i]) as++;
        else if (a[i] < b[i]) bs++;
    }

    printf("%d %d\n", as, bs);
    return 0;
}

// Diagonal difference
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, x, primary = 0, secondary = 0;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &x);
            if (i == j) primary += x;
            if (i + j == n - 1) secondary += x;
        }
    }

    printf("%d\n", abs(primary - secondary));
    return 0;
}

// Non Divisible Subset
#include <stdio.h>

int main() {
    int n, k, total = 0;
    scanf("%d %d", &n, &k);

    int count[k];
    for (int i = 0; i < k; i++) count[i] = 0;

    for (int i = 0, x; i < n; i++) {
        scanf("%d", &x);
        count[x % k]++;
    }

    total += (count[0] > 0);

    for (int i = 1; i <= k / 2; i++) {
        if (i == k - i) total += (count[i] > 0);
        else total += (count[i] > count[k - i]) ? count[i] : count[k - i];
    }

    printf("%d\n", total);
    return 0;
}

// Counter Game
#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        unsigned long long n;
        int moves = 0;
        scanf("%llu", &n);

        while (n > 1) {
            if ((n & (n - 1)) == 0) {
                n >>= 1;
            } else {
                n -= 1ULL << (63 - __builtin_clzll(n));
            }
            moves++;
        }

        printf("%s\n", (moves & 1) ? "Louise" : "Richard");
    }

    return 0;
}

// Bear and Steady
#include <stdio.h>

int id(char c) {
    if (c == 'A') return 0;
    if (c == 'C') return 1;
    if (c == 'G') return 2;
    return 3; // T
}

int main() {
    int n;
    scanf("%d", &n);

    char s[200001];
    scanf("%s", s);

    int cnt[4] = {0, 0, 0, 0};
    for (int i = 0; i < n; i++) cnt[id(s[i])]++;

    int req = n / 4;
    int left = 0, ans = n;

    for (int right = 0; right < n; right++) {
        cnt[id(s[right])]--;

        while (cnt[0] <= req && cnt[1] <= req && cnt[2] <= req && cnt[3] <= req) {
            int len = right - left + 1;
            if (len < ans) ans = len;
            cnt[id(s[left])]++;
            left++;
        }
    }

    printf("%d\n", ans);
    return 0;
}

// Caesar Cipher
#include <stdio.h>

int main() {
    int n, k;
    scanf("%d", &n);

    char s[101];
    scanf("%100s", s);
    scanf("%d", &k);

    k %= 26;

    for (int i = 0; i < n; i++) {
        if (s[i] >= 'a' && s[i] <= 'z') {
            s[i] = 'a' + (s[i] - 'a' + k) % 26;
        } else if (s[i] >= 'A' && s[i] <= 'Z') {
            s[i] = 'A' + (s[i] - 'A' + k) % 26;
        }
    }

    printf("%s\n", s);
    return 0;
}

// Panagrams
#include <stdio.h>

int main() {
    char s[10000];
    int seen[26] = {0}, count = 0;

    fgets(s, sizeof(s), stdin);

    for (int i = 0; s[i] != '\0'; i++) {
        if (s[i] >= 'A' && s[i] <= 'Z') {
            int x = s[i] - 'A';
            if (!seen[x]) { seen[x] = 1; count++; }
        } else if (s[i] >= 'a' && s[i] <= 'z') {
            int x = s[i] - 'a';
            if (!seen[x]) { seen[x] = 1; count++; }
        }
    }

    if (count == 26) printf("pangram");
    else printf("not pangram");

    return 0;
}

// Closest Numbers  
#include <stdio.h>
#include <stdlib.h>

int cmp(const void *x, const void *y) {
    return (*(int *)x - *(int *)y);
}

int main() {
    int n, a[1000];
    scanf("%d", &n);

    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    qsort(a, n, sizeof(int), cmp);

    int min = a[1] - a[0];
    for (int i = 1; i < n - 1; i++) {
        int d = a[i + 1] - a[i];
        if (d < min) min = d;
    }

    for (int i = 0; i < n - 1; i++) {
        if (a[i + 1] - a[i] == min) printf("%d %d ", a[i], a[i + 1]);
    }

    return 0;
}

// Icecream Parlor
#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int m, n;
        scanf("%d %d", &m, &n);

        int a[10000];
        for (int i = 0; i < n; i++) scanf("%d", &a[i]);

        int found = 0;
        for (int i = 0; i < n - 1 && !found; i++) {
            for (int j = i + 1; j < n; j++) {
                if (a[i] + a[j] == m) {
                    printf("%d %d\n", i + 1, j + 1);
                    found = 1;
                    break;
                }
            }
        }
    }

    return 0;
}
